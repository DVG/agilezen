def project_stories(project_id, options={})
  response_body = nil
  begin
    response = connection.get do |req|
      req.url "/api/v1/project/#{project_id}/stories", options
    end
    response_body = response.body
  rescue MultiJson::DecodeError => e
    #p 'Unable to parse JSON.'
  end
  
  response_body
end
